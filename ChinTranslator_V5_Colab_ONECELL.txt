# --- 0. Config ---------------------------------------------------------
SMOKE = False        # True = fast dry run (tiny data, 1 epoch) to test the pipeline
EPOCHS = 10          # max epochs — early stopping usually ends the run sooner
BRANCH = "claude/latest-result-5kTMp"   # repo branch to pull the code from
DRIVE_ROOT = "/content/drive/MyDrive/ChinTranslator"

import time
RUN_TAG = time.strftime("%Y%m%d-%H%M%S")
print(f"Config: SMOKE={SMOKE}  EPOCHS={EPOCHS}  RUN_TAG={RUN_TAG}")

# --- 1. GPU check (LoRA 8-bit needs a CUDA GPU) ------------------------
import torch
assert torch.cuda.is_available(), (
    "No GPU! Runtime ▸ Change runtime type ▸ Hardware accelerator ▸ GPU (T4), then Run all."
)
print("✓ GPU:", torch.cuda.get_device_name(0))

CT2_OUT  = f"{RUN_DIR}/whisper-cnh-turbo-ct2"    # faster-whisper model (persisted)
os.makedirs(RUN_DIR, exist_ok=True)
print("📁 Artifacts will be saved under:", RUN_DIR)

limit_arg = "--limit 80" if SMOKE else ""
!python prepare_data.py --out data/cv_cnh {limit_arg}

ep = 1 if SMOKE else EPOCHS
print(f"sanity: SMOKE={SMOKE}  epochs={ep}")

limit_eval = "--limit 20" if SMOKE else ""

# --- 7. Publish: model → HF model repo, app → realtime Space -----------
import os
MODEL_REPO = f"{os.environ.get('HF_USER', 'bsantisi')}/whisper-cnh-turbo-ct2"
SPACE_REPO = f"{os.environ.get('HF_USER', 'bsantisi')}/chin-realtime"

# Colab secrets are only readable from the notebook process — hoist the
# write token into the environment so the !python subprocess can see it.
if not (os.environ.get("HF_WRITE") or os.environ.get("HF_TOKEN")):
    from google.colab import userdata
    for _k in ("HF_WRITE", "HF_TOKEN"):
        try:
            _v = userdata.get(_k)
        except Exception:
            _v = None
        if _v:
            os.environ[_k] = _v
            break

# 1) model weights (+ chin_metadata.json, so the Space forces the right
#    language token automatically). Halt loudly if this fails — deploying
#    the app without the new model silently keeps serving old weights.
_rc = os.system(f'python hf_space/upload_model.py --model-dir "{CT2_OUT}" --repo "{MODEL_REPO}"')
assert _rc == 0, "model upload FAILED — fix it before deploying the app (see output above)"

# 2) the realtime app code (uses this run's branch)
os.environ["CT_REPO_DIR"] = "/content/ChinTranslator"
os.environ["CT_BRANCH"] = BRANCH
%run hf_space/deploy_colab.py

# 3) restart — required when only the MODEL repo changed (no code commit →
#    no auto-rebuild → the Space would keep serving the old weights)
import sys
sys.path.insert(0, "hf_space")
from upload_model import _resolve_hf_token
from huggingface_hub import HfApi
HfApi(token=_resolve_hf_token()).restart_space(SPACE_REPO)
print(f"\n✅ live: https://huggingface.co/spaces/{SPACE_REPO}")
print(f"   (Space variable CHIN_MODEL must be: {MODEL_REPO})")
